Coding Task 2 - Encapsulation

            
                class User2 {  
                    constructor() {  
                        this._firstName = "";  
                        this._lastName = "";  
                    }  
        
                    // Getter   
                    get firstName() {  
                        return this._firstName;  
                    }  
        
                    // Setter 
                    set firstName(firstName) {  
                        this._firstName = firstName;  
                    }  
        
                    // Getter  
                    get lastName() {  
                        return this._lastName;  
                    }  
        
                    // Setter  
                    set lastName(lastName) {  
                        this._lastName = lastName;  
                    }  
        
                    hello() {  
                        return "Hello World!";  
                    }  
                }  
        
                const user = new User2();
        
                // Set method

                user.lastName = "Doe";  
          
                const fullName = user.firstName + " " + user.lastName;  
        
                document.getElementById("q2").innerHTML = user.hello() + "<br>" + "My name is " + fullName;

Coding Task 3 - Inheritance (Subclass and Superclass)

            
                class User3 {
                    constructor() {
                        this._username = "";
                    }
    
                    // Getter
                    get username() {
                        return this._username;
                    }
    
                    // Setter
                    set username(username) {
                        this._username = username;
                    }
                }
    
                class Admin extends User3 {
                    constructor(username) {
                        super(); 
    
                        // Setter
                        this.username = username; 
                    }
    
                    expressYourRole() {
                        return "Admin";
                    }
    
                    sayhello() {
                        return "Hello " + this.expressYourRole() + " " + this.username; 
                    }
                }
    
                const admin = new Admin();
    
                // Set method
                admin.username = "Balthazar";
    
                document.getElementById("q3").innerHTML = admin.sayhello();

Coding Task 4 - Polymorphism

            
                class User4 {
                    constructor() {
                        this._numberOfArticles = 0;
                    }
        
                    // Getter
                    get numberOfArticles() {
                        return this._numberOfArticles;
                    }
        
                    // Setter
                    set numberOfArticles(numberOfArticles) {
                        this._numberOfArticles = numberOfArticles;
                    }
        
                    // Method calcScores
                    calcScores() {
                        return 0; 
                    }
                }
        
                class Author extends User4 {
                    constructor(numberOfArticles) {
                        super();
                        this.numberOfArticles = numberOfArticles;
                    }
        
                    // Method calcScores
                    calcScores() {
                        return this.numberOfArticles * 10 + 20; 
                    }
                }
        
                class Editor extends User4 {
                    constructor(numberOfArticles) {
                        super();
                        this.numberOfArticles = numberOfArticles; 
                    }
        
                    // Method calcScores
                    calcScores() {
                        return this.numberOfArticles * 6 + 15;
                    }
                }
        
                const author = new Author();
                // Set method
                author.numberOfArticles = 8;
                
                const editor = new Editor(); 
                // Set method
                editor.numberOfArticles = 15;
    
                document.getElementById("q4").innerHTML = "Author score: " + author.calcScores() + "<br>" + "Editor score: " + editor.calcScores();

                    }
