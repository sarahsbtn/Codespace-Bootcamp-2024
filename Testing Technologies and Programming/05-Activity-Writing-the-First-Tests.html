Testing Technologies and Programming


Code Challenge 1

In this topic, you will learn how to write your first tests using Cypress.io.


Testing Task 1

      
        describe('Cypress Example Tests', () => {
          it('queries an element, interacts with it and assert about the content', () => {
            cy.visit('https://example.cypress.io')
            cy.get('.home-list').contains('Traversal')
            cy.get('.home-list').contains('Traversal').click()
            cy.url().should('include', '/commands/traversal')
            cy.get('h4').should('contain', '.children()')
          })
        })

Testing Task 2

      
        describe('Cypress Example Tests', () => {
          it('queries email input, types email and assert about the content ', () => {
            cy.visit('https://example.cypress.io/commands/actions')
            cy.get('input[type="email"]').type('email@email.com');
            cy.get('input[type="email"]').should('have.value', 'email@email.com');
            });
        });


Testing Task 3

      
        describe('Cypress Example Tests', () => {
          it('query for action button and click, query for #action-canvas and click, click topLeft, click bottomRight', () => {
            cy.visit('https://example.cypress.io/commands/actions')
            cy.get('.action-btn').click();
            cy.get('#action-canvas').click();
            cy.get('#action-canvas').click('topLeft');
            cy.get('#action-canvas').click('bottomRight');
          });
        });
